    patch_tags = get_commit_tags(repo_path, patch_commit)
    introducing_tags = get_commit_tags(repo_path, introduceing_commit)
    
    vuln_tags = []
    for i in introducing_tags:
        if i not in patch_tags:
            vuln_tags.append(i)

    return vuln_tags
